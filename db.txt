

/* DATA BASE INFO */

CREATE TABLE `item_board` (
    `item_board_id` INT(11) UNSIGNED NOT NULL AUTO_INCREMENT,
    `user_id` INT(11) UNSIGNED NOT NULL ,
    `item_id` INT(11) UNSIGNED NOT NULL ,
    `item_state` INT(11) ,
    `item_board_comments` VARCHAR(1000)  COLLATE 'utf8_unicode_ci',
    `insert_time` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `update_time` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (`item_board_id`)
)
COLLATE='utf8_unicode_ci'
ENGINE=InnoDB
ROW_FORMAT=COMPACT
AUTO_INCREMENT=1
;


CREATE TABLE `item` (
    `item_id` INT(11) UNSIGNED NOT NULL AUTO_INCREMENT,
    `user_id` INT(11) UNSIGNED NOT NULL ,
    `item_title` VARCHAR(50)   COLLATE 'utf8_unicode_ci',
    `item_comments` VARCHAR(1000)  COLLATE 'utf8_unicode_ci',
    `del_yn` VARCHAR(1),
    `insert_time` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `update_time` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `delete_time` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,    
    PRIMARY KEY (`item_id`)
)
COLLATE='utf8_unicode_ci'
ENGINE=InnoDB
ROW_FORMAT=COMPACT
AUTO_INCREMENT=1
;

CREATE TABLE `item_state` (
    `item_state_id` INT(11) UNSIGNED NOT NULL AUTO_INCREMENT,    
    `item_state_title`  VARCHAR(50)  COLLATE 'utf8_unicode_ci',
    `item_statecomments` VARCHAR(1000)  COLLATE 'utf8_unicode_ci',
    `del_yn` VARCHAR(1)  NOT NULL DEFAULT 'N',
    `insert_time` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `update_time` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `delete_time` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,    
    PRIMARY KEY (`item_state_id`)
)
COLLATE='utf8_unicode_ci'
ENGINE=InnoDB
ROW_FORMAT=COMPACT
AUTO_INCREMENT=1
;
